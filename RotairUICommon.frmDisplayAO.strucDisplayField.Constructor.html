Remarks

This structure requires all its members to be useful. This constructor is just an alternate way to set them.

Example

This example displays the sales order from the Sales Order Card. It will display a field from the master (parent) and 2 fields from the line and 2 from the delivery. It uses the constructor to set the fields

VB 
Dim frmAO As New frmDisplayAO
Dim SalesOrderField As strucDisplayField
Dim SalesOrderFields() As strucDisplayField


ReDim SalesOrderFields(4)

SalesOrderField = New strucDisplayField(1, "AlphaID", imsAO_SalesOrder_CUSTOM.FieldNames.fSOM_SalesOrderID)
SalesOrderFields(0) = SalesOrderField
SalesOrderField = New strucDisplayField(2, "Line", imsAO_SalesOrderLine_CUSTOM.FieldNames.fSOI_SOLineNbr)
SalesOrderFields(1) = SalesOrderField
SalesOrderField = New strucDisplayField(2, "Item ID", imsAO_SalesOrderLine_CUSTOM.FieldNames.flSOI_IMA_ItemID)
SalesOrderFields(2) = SalesOrderField
SalesOrderField = New strucDisplayField(3, "Quantity", imsAO_SalesOrderDelivery_CUSTOM.FieldNames.fSOD_RequiredQty)
SalesOrderFields(3) = SalesOrderField
SalesOrderField = New strucDisplayField(3, "Date", imsAO_SalesOrderDelivery_CUSTOM.FieldNames.fSOD_RequiredDate)
SalesOrderFields(4) = SalesOrderField

With frmAO
    .AO = bdrSalesOrder.AO
    .DisplayField = SalesOrderFields
    .TitlePrefix = "Sales Order"
    .ShowAO()
End With
