Public Sub New( _
    Optional ByVal processNow  As Boolean = True
    ) As imsReturnData

     Dim intRevisionNumber As Integer() { _
     Dim r2d2 As imsReturnData


    'Only set any change tracking if any fields were edited.
    If Me.DocumentChanged Then
        'Set the last change date for change tracking.
        r2d2.Merge(SetValue(True, fLastRevDate, Now))
        'It is bad form to return in the middle of a function.
        '    It was done here only because this is the technique Intuitive uses.
        '    To deviate from the 'standard' set by the vendor would introduce
        '    more uncertaintity into a product which was past its uncertainity quota.
        If r2d2.AnyAppErrs Then Return r2d2

        'Update the revision number and set the type of record to 'Changed' only if this.
        '    is not a new record.
        If Me.DocumentChanged <> aoObjectState.NEW_DOCUMENT Then
            r2d2.Merge(SetValue(True, fRecType, Constants.RecType_ChangedRecord))
            intRevisionNumber = GetInt32(fRevNbr)
            r2d2.Merge(SetValue(True, fRevNbr, intRevisionNumber + 1))
        End If

        'Call the base where actual saving is done.
        r2d2.Merge(MyBase.Save(processNow))

    End If

    Return r2d2

End Function
