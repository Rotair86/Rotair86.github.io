Example

This is how to launch the party screening form for a customer.

VB 
Dim frmScreen As New frmDialogScreening


With frmScreen
    .Caller = Me.Name
    .PartyType = Enumerations.ePartyType.Customer
    'Either PartyID or PartyRecordID needs to be specified but not both.
    .PartyID = strCustomerID '<- strCustomerID holds a valid customer ID.
    .PartyRecordID = guidCustomerID '<- guidCustomerID holds a valid customer record ID.
    .ShowDialog(Me)
End With
