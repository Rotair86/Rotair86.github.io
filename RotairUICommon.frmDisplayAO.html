Example

This example displays the sales order from the Sales Order Card. It will display a field from the master (parent) and 2 fields from the line and 2 from the delivery.

VB 
Dim frmAO As New frmDisplayAO
Dim strFields() As String


ReDim strFields(4)
strFields(0) = "1AlphaID-SOM_SalesOrderID"
strFields(1) = "2Line-SOI_SOLineNbr"
strFields(2) = "2Item ID-SOI_IMA_ItemID"
strFields(3) = "3Quantity-SOD_RequiredQty"
strFields(4) = "3Date-SOD_RequiredDate"

With frmAO
    .AO = AO
    .DisplayMember = strFields
    .TitlePrefix = "Sales Order"
    .ShowAO()
End With
