Countries Method

Retiurns a dataset of countries.

Syntax
VB 
'Declaration
Public Function Countries( _
) As DataSet

Return Value

Type: System.Data.DataSet
A collection of countries.
Remarks

Returns a dataset of countries.

Example

The example populates an intuitive combobox with countries from the database.

VB 

Dim dsCountry As DataSet
Dim imsDisplayField As IntuitiveForms.Binding.imsFormattedFieldBinder
Dim imsListViewColumn As imsColumnHeader


dsCountry = RotairAO_Country.imsAO_Country.Countries()
With cboCountries
    imsListViewColumn = New imsColumnHeader
    With imsListViewColumn
        .ValueBinder.FieldName = "Country"
        .Text = "Country"
        .Width = 200
    End With
    .ListViewColumns.Add(imsListViewColumn)

    imsDisplayField = New IntuitiveForms.Binding.imsFormattedFieldBinder
    With imsDisplayField
        .FieldName = "Country"
    End With
    .Display.Fields.Add(imsDisplayField)

    'This property is required to be set
    .KeyField = "Country"

    .ListView.DataSource = dsCountry.Tables(0)

End With
